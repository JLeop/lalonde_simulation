4.2 Datensimulation¶

In einem nchsten Schritt werden Daten mit gleichartigen Eigenschaften wie die ursprnglichen Beobachtungen simuliert und dabei 50 neue Dataframes erstellt. Wir verwenden dabei einen objektorientierten Programmierungsansatz und definieren die neue Klasse Monte_Carlo, die einen Dataframe sowie die abhngige Variable als Input-Parameter verlangt. Innerhalb dieser Klasse gibt es insgesamt 6 Methoden:
  1. dummies()
    Die Methode prft, bei welchen Zeilen es sich um Dummy-Variablen handelt und gibt diese in einer Liste zurck.
  2. dummy_groups()
    Die Methode ruft eine Liste mit allen mglichen Kombinationen der Ausprgungen der Dummy-Variablen ab.
  3. indices()
    Die Methode bestimmt die Zeilenindexe, in welchen die Dummy-Variablen eine bestimmte Kombination an Ausprgungen aufweisen. Fr jede Kombinationsgruppen werden diese Indexe in einer Liste gespeichert und diese Listen wiederum in einer Liste zusammengefasst.
  4. means()
    Die Methode berechnet fr alle Beobachtungen einer Kombinationsgruppe den Durchschnittswert der unabhngigen Nicht-Dummy-Variablen. Fr dieses Projekt sind dies die Durchschnittswerte der Variablen age, educ und RE75.
  5. covariances()
    Die Methode macht dasselbe wie die Methode means(), nur berechnet sie die Kovarianzen der unabhngigen Nicht-Dummy-Variablen pro Kombinationsgruppe.
  6. monte_carlo()
    Mithilfe der vorherigen fnf Methoden fhrt nun monte_carlo() die Monte Carlo Simulation durch. Der Default Wert fr die Anzahl zu simulierenden Dataframes wurde dabei auf 50 gesetzt.
Die erstellte Klasse wird nun in der Datei Monte_Carlo.py im gleichen Arbeitsverzeichnis wie diese Jupyter Notebook gespeichert.

%%file Monte_Carlo.py

import numpy as np
import random,time
import ipywidgets as widgets

class Monte_Carlo:
    """
    Monte_Carlo simuliert Daten mit dem selben Durchschnitt
    und der selben Kovarianz wie die eingegebenen Daten
    """
    def __init__(self, df, y):
        self.df = df
        # Liste aller unabhngier Variablen ohne Dummies
        variables = [variable for variable in list(df.columns.values) if variable not in ["treated", "no-deg", "married"] + ["RE78"]]
        # Spaltenindex der ersten unabhngier Variablen ohne Dummies
        self.start = df.columns.get_loc(variables[0])
        # Spaltenindex der letzten Variablen
        self.stopp = self.start + len(variables)
        
    def dummies(self):
        no_dummies = []
        dummies = []
        # Ruft alle Spaltennamen ab
        columns = self.df.columns
        # Findet alle Spalten die KEINE Dummy Variablen enthalten
        for column in columns:
            no_dummies += no_dummies + [column for cell in self.df[column] if cell != 1.0 and cell != 0.0]
        no_dummies = set(no_dummies)
        # Findet alle Spalten die eine Dummy Variabel enthlt
        for i in self.df.columns:
            if i not in no_dummies:
                dummies.append(i)
        return dummies

    def dummy_groups(self):
        groups = []
        # Ruft die Vorherige Methode auf um alle Dummy Spalten zu bekommen
        dummies = self.dummies()
        # Macht fr jede Zeile einen Vektor, der aus den Ausprgungen aller Dummy Variablen besteht
        # und fasst all diese Vektoren in einer List zusammen
        all_groups = [row for row in self.df[dummies].as_matrix().astype(int).tolist()]
        # Macht aus der Vektor-Liste eine neue Liste, in welchem jeder Vektor einmalig ist
        for group in all_groups:
            if group not in groups:
                groups.append(group)
        return groups
    
    def indices(self):
        # Kreiert eine Liste mit Unterlisten, welche die Indexe fr jede Dummy-Gruppe enthalten
        indices = []
        # Kreiert eine Liste aller unabhngigen Variablen ohne Dummies
        dummies = self.dummies()
        for i in self.dummy_groups():
            # Dieser Schritt knnte Mithilfe der dummies() Methode veralgemeinert werden.
            # Do this as list comprehension
            index = self.df[ ( self.df[dummies[0]] == i[0] ) & ( self.df[dummies[1]] == i[1] ) & ( self.df[dummies[2]] == i[2] ) ].index
            indices.append(index)
        return indices

    # Berechnet die Durchschnitte fr jede Dummy-Gruppe
    def means(self):
        indices = self.indices()
        means = []
        for i in indices:
            mean = self.df.iloc[ i, self.start : self.stopp ].mean()
            means.append(mean)
        return means
        
    # Berechnet die Kovarianzen fr jede Dummy-Gruppe
    def covariances(self):
        covs = []
        indices = self.indices()
        for i in indices:
            cov = self.df.iloc[ i, self.start : self.stopp ].cov()
            covs.append(cov)
        return covs
    
    def monte_carlo(self, iterations = 50):
        # Zuvor definierte Methoden werden abgerufen
        indices = self.indices()
        means = self.means()
        covariances = self.covariances()
        df = self.df
        dummy_groups = self.dummy_groups()
        
        # Widget fr Ladebalken
        w = widgets.IntProgress(
        value=0,
        min=0,
        max=iterations,
        step=1,
        description='Loading:',
        bar_style='success',
        orientation='horizontal'
        )
        display(w)
        
        # Eine leere Liste in der die simulierten dfs angefgt werden.
        mc_dfs = []
        # Standardmssig werden 50 Iterationen durchgefhrt
        for i in range(0,iterations):
            # Originales df wird kopiert
            df_MC = df.copy()
            # Fr jede der 8 Dummy-Gruppen werden die Werte durch Simulationen ersetzt, 
            # welche den Mittelwert und die Kovarianz bercksichtigen.
            for j in range(0,len(dummy_groups)):
                df_MC.iloc[ indices[j], self.start : self.stopp ] = np.random.multivariate_normal( means[j], covariances[j], len( indices[j] ) )        
            mc_dfs.append(df_MC)
            w.value = w.value + 1
        w.close()

        return mc_dfs

4.3 Implementierung des Schätzverfahrens¶

4.3.1 Algorithmus zur Schätzung des Effekts der Teilnahme am Beschäftigungsprogramm (inklusive Standardfehler)¶

In einem nchsten Schritt wird der Effekt des Beschftigungsprogrammes auf die Teilnehmer untersucht. Dazu verwenden wir die Schtzstatistik $\hat{\theta}$. $$ \hat{\theta} = \frac{1}{N}\sum^{N}_{i = 1}(\mu(1) - \mu(1)) $$
Die Schtzstatistik wurde unter Verwendung einer OLS-Regression berechnet. Wobei $\mu(1)$ der geschtzten abhngien Variable RE78 entspricht. $\mu(1)$ wurde mit den Koeffizienten aus einer Regression mit nur Teilnehmern geschtzt. Diese Regression kann wie folgt beschrieben werden:

$$ \rm\small \begin{align} \begin{bmatrix} RE78_{1} \\ RE78_{2} \\ RE78_{3} \\ \vdots \\ RE78_{156} \\ \end{bmatrix} = \begin{bmatrix} 1 & treated_{1} & married{1} & no-deg_{1} & age_{1} & educ_{1} & RE75_{1}\\ 1 & treated_{2} & married{2} & no-deg_{2} & age_{2} & educ_{2} & RE75_{2}\\ 1 & treated_{3} & married{3} & no-deg_{3} & age_{3} & educ_{3} & RE75_{3}\\ \vdots & \vdots & \vdots & \vdots & \vdots & \vdots & \vdots \\ 1 & treated_{156} & married_{156} & no-deg_{156} & age_{156} & educ_{156} & RE75_{156}\\ \end{bmatrix} \times \begin{bmatrix} \beta_{0} \\ \beta_{1} \\ \beta_{2} \\ \vdots \\ \beta_{6} \\ \end{bmatrix} + \begin{bmatrix} u_{1} \\ u_{2} \\ u_{3} \\ \vdots \\ u_{156} \\ \end{bmatrix} \end{align} $$
$\mu(0)$ wiederum sind die geschtzten RE78 aus einer Regression mit nur Nichtteilnehmern. Die Schtzstatistik nimmt nun die durchschnittliche Differenz zwischen geschtztem Teilnehmerlohn und geschtztem Nichtteilnehmerlohn im Jahr 1978 an.

Um spter Konfidenzintervalle berechnen zu knnen, bentigt man die Standardfehler der Schtzstatistiken. Zur Berechnung dieser verwenden wir die Standard-Bootstrapmethode und imitieren 19 Replikationen der ursprnglichen Daten. Aus jeder dieser Replikationen wird die Teststatistik $\hat{\theta}_{b}$ berechnet und schliesslich der Standardfehler $\sigma(\hat{\theta})$ aus den 19 Teststatistiken ermittelt.
Wir erstellen zuerst die Funktion boostrap(), da diese in der nachfolgenden Klasse verwendet wird. boostrap() nimmt einen Dataframe, auf den die Bootstrap-Methode angewendet werden soll, als Input-Parameter. Daneben kann auch noch die Anzahl an Bootstrap Replikationen festgelegt werden. Standardmssig ist dieser Wert auf 19 Replikationen gesetzt.

Durch Ausfhren der Funktion werden zufllig Beobachtungen aus dem eingegeben Dataframe genommen und diese zu einem neuen Dataframe zusammengesetzt. Dadurch entsteht die geforderte Anzahl an replizierten Datenstzen. Diese Funktion wird im selben Arbeitsverzeichnis wie dieses Jupyter Notebok in der Datei bootstrap.py abgespeichert.

%%file bootstrap.py

import numpy as np

def bootstrap(df, B = 19):
    """
    Die Funktion erzeugt eine definierte Anzahl Bootstrap Replikationen
    fr ein gegebenes Pandas Dataframe
    """
    df_bootstrapped_list = []
    for i in range(0,B):
        # Zeilen mit zuflligen Indexzahlen Beobachtungen auswhlen und neues df daraus erstellen 
        df_random = df.loc[np.random.randint(0, df.shape[0], df.shape[0]), :]
        df_bootstrapped_list.append(df_random)
    return df_bootstrapped_list

Overwriting bootstrap.py
Zur Berechnung der empirischen Teststatistik wenden wir erneut die objektorientierte Programmierung an. Wir definieren die Klasse Test_Statistics(). Diese Klasse bentigt einen Dataframe, eine Dummy-Variable als Vergleich (hier treated), eine abhngige Variable (hier RE78), sowie eine unbestimmte Anzahl an unabhngigen Variablen.

Zuerst wird das eingegebene Dataframe mit einer Spalte bestehend aus Einsen ergnzt. Diese Spalte wird bentigt fr die berechnung des Achsenabschnittsparameters. Danach werden die Berechnungen in den einzelnen Methoden durchgefhrt:
  1. Die Methoden 1 - 9 berechnen die einzelnen Elemente der Schtzstatistik. Auf diese Weise kann nicht nur die Schtzstatistik als Ganzes, sonder auch ihre einzelnen Komponenten abgerufen werden.
  2. Die Methode theta_hat() berechnet die Schtzstatistik $\hat{\theta}$.
  3. Die Methode standard_error_boot() beinhaltet die zuvor definierte Bootstrap Funktion. In einem ersten Teil werden die Bootstrap Replikationen der empirischen Daten erzeugt. Standardmssig werden 19 Replikationen induziert. Danach wird fr jede Replikation die Schtzstatistik $\hat{\theta}$ berechnet und aus diesen Schtzstatistiken der Standardfehler ermittelt.
Auch diese Funktion wird in einer Datei im selben Arbeitsverzeichnis wie dieses Jupyter Notebook abgespeichert.

%%file Test_Statistics.py

from bootstrap import bootstrap
import numpy as np
import math

class Test_Statistics:
    """
    Berechnet die Schtzstatistik Theta-Dach sowie
    den Standardfehler einer Bootstrap-Replikation.
    """

    def __init__(self, df, compare, y, *x):
        # Wir verwenden einen Klon damit das originale df erhalten bleibt
        self.df = df.copy()
        # Spalte mit Einsen an das df anfgen.
        self.df["ones"] = np.ones(self.df.shape[0])
        self.compare = compare
        self.y = y
        self.x = x
        # Spalte aller unabhngigen Variablen und des Achsenabschnittparameters.
        self.columns = list(x) + ["ones"]
        
    def X(self): # Macht aus den X und nicht den Y Zeilen eine Matrix
        X = self.df[self.columns].as_matrix()
        return X
        
    def X1(self):
        # Teilmatrix von Teilnehmern
        X1 = self.df.loc[self.df[self.compare] == 1][self.columns].as_matrix()
        return X1
    
    def X0(self):
        # Teilmatrix von nicht Teilnehmern
        X0 = self.df.loc[self.df[self.compare] == 0][self.columns].as_matrix()
        return X0
    
    def y1(self):
        # Einkommen 1978 von Teilnehmern
        y1 = self.df.loc[self.df[self.compare] == 1][self.y].as_matrix()
        return y1
    
    def y0(self):
        # Einkommen 1978 von nicht Teilnehmern
        y0 = self.df.loc[self.df[self.compare] == 0][self.y].as_matrix()
        return y0
    
    def b1(self):
        X1 = self.X1()
        y1 = self.y1()
        # Schtzung der Steigungsparameter nur mit Teilnehmern
        b1 = np.dot(np.linalg.inv(np.dot(X1.T, X1)), np.dot(X1.T, y1))
        return b1
    
    def b0(self):
        X0 = self.X0()
        y0 = self.y0()
        # Schtzung der Steigungsparameter nur mit Nichtteilnehmern
        b0 = np.dot(np.linalg.inv(np.dot(X0.T, X0)), np.dot(X0.T, y0))
        return b0
    
    def mu_1(self):
        X = self.X()
        b1 = self.b1()
        # Schtzung Einkommen 1978 mit Koeffizienten von Teilnehmern
        mu_1 = np.dot(X, b1)
        return mu_1
    
    def mu_0(self):
        X = self.X()
        b0 = self.b0()
        # Schtzung Einkommen 1978 mit Koeffizienten von Nichtteilnehmern
        mu_0 = np.dot(X, b0)
        return mu_0
    
    def theta_hat(self):
        X = self.X()
        b1 = self.b1()
        b0 = self.b0()
        # Berechnung des Effekts Theta-Dach
        theta_hat = (np.dot(X, b1) - np.dot(X, b0)).mean()
        return theta_hat
    
    def standard_error_boot(self, B = 19):
        # Fhrt die bootstrap() Funktion aus und erzeugt default-mssig 19 Replikationen
        df_b_list = bootstrap(self.df, B)
        # Schtzstatistik Theta-Dach fr jede Replikation berechnen
        theta_hat_b_list = [Test_Statistics(df_b, self.compare, self.y, *self.x).theta_hat() for df_b in df_b_list]
        #Standardfehler der Schtzstatistiken berechnen
        std_theta_hat = np.std(theta_hat_b_list)
        return std_theta_hat

4.3.2 Algorithmus zur Schätzung simulierten Effekte inklusive Standardfehler¶

Das gleiche Vorgehen wie unter Abschnitt 4.3.1 wird nun auch fr jeden in der Monte Carlo Simulation generierten Dataframe durchgefhrt. Erneut werden der Effekt mithilfe der Schtzstatistik $\hat{\theta}_{sim}$ sowie der Standardfehler $\hat{\sigma}(\hat{\theta}_{sim})$ durch die Bootstrap-Methode berechnet.

Da sich die Schtzstatistik der simulierten Daten $\hat{\theta}_{sim}$ von jener der empirischen Daten unterscheidet, definieren wir eine neue Klasse Test_Statistics_Sim(), welche die abgenderte Schtzstatistik berechnet. Dazu knnen wir die verwendeten Methoden aus der Klasse Test_Statistics() erben, sodass diese auch in der neuen Klasse verfgbar sind.

Mithilfe der geerbten Attribute und Methoden, die bereits in Test_Statistics() definiert wurden, berechnen wir alle notwendigen Variablen, um die Gleichungen 7 - 9 zu berechnen.

Gleichungen 8 und 9, zur Berechnung von $\mu_{sim}(1)$ und $\mu_{sim}(0)$ werden durch die Methoden .mu_sim() und .mu_sim0() definiert. Mithilfe dieser Methoden definieren wir Gleichung 7, zur Berechnung von $\hat{\theta}_{sim}$ als Methode .theta_hat_sim()

Zuletzt wird wieder mithilfe der Bootstrap Methode der Standardfehler $\hat{\sigma}(\hat{\theta}_{sim})$ durch die Methode standard_error_sim_boot() berechnet, welche die vernderte Berechnung von $\hat{\theta}_{sim}$ bercksichtigt.

%%file Test_Statistics_Sim.py

import numpy as np
import math

# Funktionen die in der standard_error Methode gebraucht werden importieren.
from bootstrap import bootstrap
from Test_Statistics import Test_Statistics

class Test_Statistics_Sim(Test_Statistics):
    """
    Berechnet die angepasste Schtzstatistik Theta-Dach fr Simulationsdaten sowie
    den Standardfehler einer Bootstrap-Replikation von diesen Daten.
    """
    def __init__(self, df, compare, y, *x):
        super().__init__(df, compare, y, *x)
         # Attribute fr die simulierten Dataframes
        self.N = self.df.shape[0]
        self.u = [np.random.normal() for i in range(0,self.N)]
        self.v = [np.random.normal() for i in range(0,self.N)]
        
    def K(self):
        # Spalten von X
        K = self.X().shape[1]
        return K
    
    def N1(self):
        # Anzahl Teilnehmer
        N1 = len(self.df.loc[self.df[self.compare] == 1])
        return N1
    
    def N0(self):
        # Anzahl Nichtteilnehmer
        N0 = len(self.df.loc[self.df[self.compare] == 0])
        return N0
    
    def ssr1(self):
        # Sum of Squared Residuals aus der Regression mit nur Teilnehmer
        ssr1 = sum((np.dot(self.X1(),self.b1())-self.y1())**2)
        return ssr1
    
    def ssr0(self):
        # Sum of Squared Residuals aus der Regression mit nur Nichtteilnehmer
        ssr0 = sum((np.dot(self.X0(),self.b0())-self.y0())**2)
        return ssr0
        
    def mu_sim1(self):
        mu_sim1 = np.dot(self.X(),self.b1()) + np.dot(self.u,math.sqrt(self.ssr1()/(self.N1()-self.K())))
        return mu_sim1

    def mu_sim0(self):
        mu_sim0 = np.dot(self.X(),self.b0()) + np.dot(self.v,math.sqrt(self.ssr0()/(self.N0()-self.K())))
        return mu_sim0
    
    def theta_hat_sim(self):
        theta_hat_sim = sum(self.mu_sim1()-self.mu_sim0())/self.N
        return theta_hat_sim

    def standard_error_sim_boot(self, B = 19):
        # Fhrt die bootstrap() Funktion aus und erzeugt default-mssig 19 Replikationen
        df_b_list = bootstrap(self.df, B)
        # Schtzstatistik Theta-Dach fr jede Replikation berechnen
        theta_hat_b_list = [Test_Statistics_Sim(df_b, self.compare, self.y, *self.x).theta_hat_sim() for df_b in df_b_list]
        #Standardfehler ausrechnen
        std_theta_hat = np.std(theta_hat_b_list)
        return std_theta_hat
